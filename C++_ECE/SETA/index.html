SET A - Q1: Single Linked List ADT Operations

1. AIM

To implement a Singly Linked List ADT with basic operations like insertion, deletion, and display.

2. ALGORITHM (PSEUDOCODE)

  1. Define Node with data and next pointer.
  2. Define LinkedList class with head initialized to NULL.
  3. Insert: Create node. If head is NULL, make it head. Else, traverse to end and link.
  4. Delete: Traverse to find key. Link previous node to next node and delete target.
  5. Display: Traverse from head to NULL, printing data.

3. C++ PROGRAM

#include <iostream>
using namespace std;
struct Node { int data; Node* next; };
class SLL {
    Node* head = NULL;
public:
    void insert(int v) {
        Node* n = new Node{v, NULL};
        if(!head) head = n; else { Node* t=head; while(t->next) t=t->next; t->next=n; }
    }
    void del(int v) {
        if(!head) return;
        if(head->data==v) { Node* t=head; head=head->next; delete t; return; }
        Node *t=head, *p=NULL;
        while(t && t->data!=v) { p=t; t=t->next; }
        if(t) { p->next=t->next; delete t; }
    }
    void disp() { for(Node* t=head; t; t=t->next) cout<<t->data<<"->"; cout<<"NULL\n"; }
};
int main() { SLL l; l.insert(10); l.insert(20); l.disp(); l.del(10); l.disp(); return 0; }

SET A - Q10: Infix to Postfix Expression using Stack

1. AIM

To convert an Infix expression to Postfix using a Stack.

2. ALGORITHM (PSEUDOCODE)

  1. If operand, output. If '(', push. If ')', pop to output until '('.
  2. If operator, pop higher/equal precedence to output, then push.

3. C++ PROGRAM

#include <iostream>
#include <stack>
using namespace std;
int prec(char c) { if(c=='^') return 3; if(c=='*'||c=='/') return 2; if(c=='+'||c=='-') return 1; return -1; }
void inToPost(string s) {
    stack<char> st; string res;
    for(char c:s) {
        if(isalnum(c)) res+=c; else if(c=='(') st.push(c);
        else if(c==')') { while(st.top()!='(') {res+=st.top(); st.pop();} st.pop(); }
        else { while(!st.empty() && prec(c)<=prec(st.top())){res+=st.top(); st.pop();} st.push(c); }
    }
    while(!st.empty()){res+=st.top(); st.pop();} cout<<res<<"\n";
}
int main() { inToPost("a+b*c"); return 0; }

SET A - Q13: AVL Tree Insertion & Deletion

1. AIM

To implement AVL Tree self-balancing insertion.

2. ALGORITHM (PSEUDOCODE)

  1. Standard BST insert.
  2. Get balance factor. If > 1 or < -1, rotate (LL, RR, LR, RL) to fix.

3. C++ PROGRAM

#include <iostream>
using namespace std;
struct Node { int d, h; Node *l, *r; };
int h(Node* n) { return n?n->h:0; } int max(int a, int b) { return a>b?a:b; }
Node* R(Node* y) { Node* x=y->l; y->l=x->r; x->r=y; y->h=max(h(y->l),h(y->r))+1; x->h=max(h(x->l),h(x->r))+1; return x; }
Node* L(Node* x) { Node* y=x->r; x->r=y->l; y->l=x; x->h=max(h(x->l),h(x->r))+1; y->h=max(h(y->l),h(y->r))+1; return y; }
int bal(Node* n) { return n?h(n->l)-h(n->r):0; }
Node* ins(Node* n, int k) {
    if(!n) return new Node{k,1,NULL,NULL};
    if(k<n->d) n->l=ins(n->l,k); else n->r=ins(n->r,k);
    n->h = 1+max(h(n->l),h(n->r)); int b = bal(n);
    if(b>1 && k<n->l->d) return R(n);
    if(b<-1 && k>n->r->d) return L(n);
    return n;
}
void in(Node* r) { if(r){ in(r->l); cout<<r->d<<" "; in(r->r); } }
int main() { Node* r=NULL; r=ins(r,10); r=ins(r,20); r=ins(r,30); in(r); return 0; }

To check if parentheses in an expression are balanced.

2. ALGORITHM (PSEUDOCODE)

  1. Traverse string. Push ({[ to stack.

  3. If stack empty at end, return true.

3. C++ PROGRAM

#include <iostream>
#include <stack>
using namespace std;
bool isValid(string s) {
    stack<char> st;
    for(char c:s) {
        if(c=='('||c=='{'||c=='[') st.push(c);
        else {
            if(st.empty()) return false;
            char t=st.top(); st.pop();
            if((c==')'&&t!='(') || (c=='}'&&t!='{') || (c==']'&&t!='[')) return false;
        }
    }
    return st.empty();
}
int main() { cout << (isValid("{[]}")?"Balanced":"Not Balanced") << "\n"; return 0; }
