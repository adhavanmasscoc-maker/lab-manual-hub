SET B - Q1: Single Linked List ADT Operations

1. AIM

To implement a Singly Linked List ADT with basic insertion and display operations.

2. ALGORITHM (PSEUDOCODE)

  1. Define Node with data and next pointer.
  2. Define LinkedList class with head initialized to NULL.
  3. Insert: Create node. If head is NULL, make it head. Else, traverse to end and link.
  4. Display: Traverse from head to NULL, printing data.

3. C++ PROGRAM

#include <iostream>
using namespace std;

struct Node
{
    int data;
    Node *next;
};

class SLL
{
    Node *head = NULL;
public:
    void insert(int v)
    {
        Node *n = new Node;
        n->data = v;
        n->next = NULL;

        if(head == NULL)
        {
            head = n;
        }
        else
        {
            Node *t = head;
            while(t->next != NULL)
            {
                t = t->next;
            }
            t->next = n;
        }
    }

    void disp()
    {
        Node *t = head;
        while(t != NULL)
        {
            cout << t->data << "->";
            t = t->next;
        }
        cout << "NULL\n";
    }
};

int main()
{
    SLL l;
    l.insert(10);
    l.insert(20);
    l.disp();
    return 0;
}

            head = n;
            n->next = head;

};

SET B - Q4: Polynomial Subtraction using Singly Linked List

1. AIM

To subtract two polynomials stored in linked lists.

2. ALGORITHM (PSEUDOCODE)

  1. Traverse both lists. If powers match, subtract coeffs.
  2. If pow1 > pow2, add node1. If pow1 < pow2, subtract node2 coeff.

3. C++ PROGRAM

#include <iostream>
using namespace std;

struct Node
{
    int c;
    int p;
    Node *next;
};

void ins(Node*& h, int c, int p)
{
    Node *n = new Node;
    n->c = c;
    n->p = p;
    n->next = NULL;

    if(h == NULL)
    {
        h = n;
    }
    else
    {
        Node *t = h;
        while(t->next != NULL)
        {
            t = t->next;
        }
        t->next = n;
    }
}

void sub(Node *h1, Node *h2)
{
    while(h1 != NULL && h2 != NULL)
    {
        if(h1->p > h2->p)
        {
            cout << h1->c << "x^" << h1->p << " ";
            h1 = h1->next;
        }
        else if(h1->p < h2->p)
        {
            cout << -h2->c << "x^" << h2->p << " ";
            h2 = h2->next;
        }
        else
        {
            int res = h1->c - h2->c;
            if(res != 0)
            {
                cout << res << "x^" << h1->p << " ";
            }
            h1 = h1->next;
            h2 = h2->next;
        }
    }
    cout << "\n";
}

int main()
{
    Node *p1 = NULL;
    Node *p2 = NULL;
    
    ins(p1, 5, 2);
    ins(p1, 4, 1);
    
    ins(p2, 3, 2);
    ins(p2, 2, 0);
    
    sub(p1, p2);
    return 0;
}

1. AIM

To implement Queue (FIFO) operations using a Linked List.

2. ALGORITHM (PSEUDOCODE)

  1. Enqueue: Add node to rear. Update rear.
  2. Dequeue: Remove node from front. Update front.

3. C++ PROGRAM

#include <iostream>
using namespace std;

struct Node
{
    int data;
    Node *next;
};

class Queue
{
    Node *f = NULL;
    Node *r = NULL;
public:
    void enq(int x)
    {
        Node *n = new Node;
        n->data = x;
        n->next = NULL;

        if(r == NULL)
        {
            f = n;
            r = n;
        }
        else
        {
            r->next = n;
            r = n;
        }
    }

    void deq()
    {
        if(f == NULL)
        {
            return;
        }
        
        Node *t = f;
        f = f->next;
        
        if(f == NULL)
        {
            r = NULL;
        }
        delete t;
    }

    void disp()
    {
        Node *t = f;
        while(t != NULL)
        {
            cout << t->data << " ";
            t = t->next;
        }
        cout << "\n";
    }
};

int main()
{
    Queue q;
    q.enq(10);
    q.enq(20);
    q.disp();
    
    q.deq();
    q.disp();
    return 0;
}

SET B - Q6: Circular Queue using Linked List

1. AIM

To implement a Circular Queue using a Linked List.

2. ALGORITHM (PSEUDOCODE)

  1. Keep a rear pointer. Front is rear->next.
  2. Enqueue: Insert after rear, link to front. Update rear.

3. C++ PROGRAM

#include <iostream>
using namespace std;

struct Node
{
    int data;
    Node *next;
};

class CQ
{
    Node *r = NULL;
public:
    void enq(int v)
    {
        Node *n = new Node;
        n->data = v;
        n->next = NULL;

        if(r == NULL)
        {
            r = n;
            r->next = r;
        }
        else
        {
            n->next = r->next;
            r->next = n;
            r = n;
        }
    }

    void deq()
    {
        if(r == NULL)
        {
            return;
        }
        
        Node *t = r->next;
        
        if(r == t)
        {
            r = NULL;
        }
        else
        {
            r->next = t->next;
        }
        delete t;
    }

    void disp()
    {
        if(r == NULL)
        {
            return;
        }
        
        Node *t = r->next;
        do
        {
            cout << t->data << " ";
            t = t->next;
        } while(t != r->next);
        
        cout << "\n";
    }
};

int main()
{
    CQ c;
    c.enq(1);
    c.enq(2);
    c.disp();

SET B - Q8: Balanced Parentheses using Stack

1. AIM

To check if parentheses in an expression are balanced.

2. ALGORITHM (PSEUDOCODE)

  1. Traverse string. Push ({[ to stack.
  2. If closing bracket, pop. If mismatch, return false.
  3. If stack empty at end, return true.

3. C++ PROGRAM

#include <iostream>
#include <stack>
using namespace std;

bool isValid(string s)
{
    stack<char> st;
    for(char c : s)
    {
        if(c == '(' || c == '{' || c == '[')
        {
            st.push(c);
        }
        else
        {
            if(st.empty())
            {
                return false;
            }
            
            char t = st.top();
            st.pop();
            
            if((c == ')' && t != '(') || (c == '}' && t != '{') || (c == ']' && t != '['))
            {
                return false;
            }
        }
    }
    return st.empty();
}

int main()
{
    if(isValid("{[]}"))
    {
        cout << "Balanced\n";
    }
    else
    {
        cout << "Not Balanced\n";
    }
    return 0;
}

SET B - Q11: Infix to Postfix Expression

1. AIM

To convert an Infix expression to Postfix using a Stack.

2. ALGORITHM (PSEUDOCODE)

  1. If operand, output. If '(', push. If ')', pop to output until '('.
  2. If operator, pop higher/equal precedence to output, then push.

3. C++ PROGRAM

#include <iostream>
#include <stack>
using namespace std;

int prec(char c)
{
    if(c == '^')
        return 3;
    if(c == '*' || c == '/')
        return 2;
    if(c == '+' || c == '-')
        return 1;
    return -1;
}

void inToPost(string s)
{
    stack<char> st;
    string res;
    
    for(char c : s)
    {
        if(isalnum(c))
        {
            res += c;
        }
        else if(c == '(')
        {
            st.push(c);
        }
        else if(c == ')')
        {
            while(st.top() != '(')
            {
                res += st.top();
                st.pop();
            }
            st.pop();
        }
        else
        {
            while(!st.empty() && prec(c) <= prec(st.top()))
            {
                res += st.top();
                st.pop();
            }
            st.push(c);
        }
    }
    
    while(!st.empty())
    {
        res += st.top();
        st.pop();
    }
    cout << res << "\n";
}

int main()
{
    inToPost("A+B*C");
    return 0;
}

#include <iostream>
using namespace std;

struct Node
{
    int data;
    Node *l;
    Node *r;
};

Node* ins(Node* r, int v)
{
    if(r == NULL)
    {
        Node *n = new Node;
        n->data = v;
        n->l = NULL;
        n->r = NULL;
        return n;

    return r;

{
    if(r != NULL)
    {
        in(r->l);
        cout << r->data << " ";
        in(r->r);
    }

    ins(r, 30);
    ins(r, 70);

SET B - Q13: B-Tree Search Function

1. AIM

To implement search logic within a B-Tree structure.

2. ALGORITHM (PSEUDOCODE)

  1. Iterate through keys in current node. If key matches, return true.
  2. If key < current key, recurse into corresponding child pointer.

3. C++ PROGRAM

#include <iostream>
#include <vector>
using namespace std;

struct BNode
{
    vector<int> keys;
    vector<BNode*> child;
    bool leaf = true;
};

bool search(BNode *r, int k)
{
    if(r == NULL)
    {
        return false;
    }
    
    int i = 0;
    while(i < r->keys.size() && k > r->keys[i])
    {
        i++;
    }
    
    if(i < r->keys.size() && k == r->keys[i])
    {
        return true;
    }
    
    if(r->leaf)
    {
        return false;
    }
    
    return search(r->child[i], k);
}

int main()
{
    BNode *r = new BNode();
    r->keys = {10, 20};
    
    if(search(r, 20))
    {
        cout << "Found\n";
    }
    else
    {
        cout << "Not Found\n";
    }
    return 0;
}

SET B - Q14: Max-Heap Operations

1. AIM

To implement Max-Heap insertion and heapify.

2. ALGORITHM (PSEUDOCODE)

  1. Add element at end. While parent < child, swap (Heapify Up).

3. C++ PROGRAM

#include <iostream>
#include <vector>
using namespace std;

class MaxHeap
{
    vector<int> h;
public:
    void push(int v)
    {
        h.push_back(v);
        int i = h.size() - 1;
        
        while(i != 0 && h[(i - 1) / 2] < h[i])
        {
            swap(h[i], h[(i - 1) / 2]);
            i = (i - 1) / 2;
        }
    }
    
    void disp()
    {
        for(int x : h)
        {
            cout << x << " ";
        }
        cout << "\n";
    }
};

int main()
{
    MaxHeap m;
    m.push(10);
    m.push(20);
    m.push(5);
    
    m.disp();
    return 0;

SET B - Q15: AVL Tree Insertion Operation

1. AIM

To implement AVL Tree self-balancing insertion.

2. ALGORITHM (PSEUDOCODE)

  1. Standard BST insert. Calculate balance factor.
  2. If > 1 or < -1, rotate (LL, RR, LR, RL) to fix.

3. C++ PROGRAM

#include <iostream>
using namespace std;

struct Node
{
    int d;
    int h;
    Node *l;
    Node *r;
};

int h(Node* n)
{
    if(n != NULL)
    {
        return n->h;
    }
    return 0;
}

int max(int a, int b)
{
    if(a > b)
    {
        return a;
    }
    return b;
}

Node* R(Node* y)
{
    Node *x = y->l;
    y->l = x->r;
    x->r = y;
    y->h = max(h(y->l), h(y->r)) + 1;
    x->h = max(h(x->l), h(x->r)) + 1;
    return x;
}

Node* L(Node* x)
{
    Node *y = x->r;
    x->r = y->l;
    y->l = x;
    x->h = max(h(x->l), h(x->r)) + 1;
    y->h = max(h(y->l), h(y->r)) + 1;
    return y;
}

int bal(Node* n)
{
    if(n != NULL)
    {
        return h(n->l) - h(n->r);
    }
    return 0;
}

Node* ins(Node* n, int k)
{
    if(n == NULL)
    {
        Node *newNode = new Node;
        newNode->d = k;
        newNode->h = 1;
        newNode->l = NULL;
        newNode->r = NULL;
        return newNode;
    }
    
    if(k < n->d)
    {
        n->l = ins(n->l, k);
    }
    else
    {
        n->r = ins(n->r, k);
    }
    
    n->h = 1 + max(h(n->l), h(n->r));
    int b = bal(n);
    
    if(b > 1 && k < n->l->d)
    {
        return R(n);
    }
    if(b < -1 && k > n->r->d)
    {
        return L(n);
    }
    
    return n;
}

void in(Node* r)
{
    if(r != NULL)
    {
        in(r->l);
        cout << r->d << " ";
        in(r->r);
    }
}

int main()
{
    Node *r = NULL;
    r = ins(r, 10);
    r = ins(r, 20);
    r = ins(r, 30);
    
    in(r);
    return 0;
}

g++ q15.cpp -o q15 && ./q15
