EX NO: 1 - Implement Single Linked List ADT Operations

Aim

To write a C++ program to implement Singly Linked List ADT operations (Insertion, Deletion, and Display).

Algorithm

  1. Define a Node structure with data and a pointer to the next node.
  2. Initialize the head pointer to NULL.
  3. Insertion: Create a new node. If the list is empty, make it the head. Otherwise, traverse to the end and link the new node.
  4. Display: Traverse from the head to NULL, printing the data of each node.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

class SinglyLinkedList {
    Node* head;
public:
    SinglyLinkedList() { head = NULL; }
    
    void insert(int val) {
        Node* newNode = new Node();
        newNode->data = val;
        newNode->next = NULL;
        if (head == NULL) {
            head = newNode;
        } else {
            Node* temp = head;
            while (temp->next != NULL) {
                temp = temp->next;
            }
            temp->next = newNode;
        }
    }
    
    void display() {
        Node* temp = head;
        while (temp != NULL) {
            cout << temp->data << " -> ";
            temp = temp->next;
        }
        cout << "NULL\n";
    }
};

int main() {
    SinglyLinkedList list;
    list.insert(10);
    list.insert(20);
    list.insert(30);
    cout << "Singly Linked List: ";
    list.display();
    return 0;
}

Output

Singly Linked List: 10 -> 20 -> 30 -> NULL

Result

Thus, the program to implement Single Linked List ADT was executed successfully and output verified.

EX NO: 2 - Implement Doubly Linked List ADT Operations

Aim

To write a C++ program to implement Doubly Linked List ADT operations.

Algorithm

  1. Define a Node structure with data, a next pointer, and a prev pointer.
  2. Insertion: Create a new node. If empty, make it head. Otherwise, link the last node's next to it, and its prev to the last node.
  3. Display: Traverse from head using the next pointer.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
    Node* prev;
};

class DoublyLinkedList {
    Node* head;
public:
    DoublyLinkedList() { head = NULL; }
    
    void insert(int val) {
        Node* newNode = new Node();
        newNode->data = val;
        newNode->next = NULL;
        if (head == NULL) {
            newNode->prev = NULL;
            head = newNode;
        } else {
            Node* temp = head;
            while (temp->next != NULL) temp = temp->next;
            temp->next = newNode;
            newNode->prev = temp;
        }
    }
    
    void display() {
        Node* temp = head;
        while (temp != NULL) {
            cout << temp->data << " <-> ";
            temp = temp->next;
        }
        cout << "NULL\n";
    }
};

int main() {
    DoublyLinkedList dll;
    dll.insert(5);
    dll.insert(15);
    dll.insert(25);
    cout << "Doubly Linked List: ";
    dll.display();
    return 0;
}

EX NO: 3 - Implement Circular Linked List ADT Operations

Aim

To write a C++ program to implement Circular Linked List ADT operations.

Algorithm

  1. Define a Node structure with data and next pointer.
  2. Insertion: Create a new node. If empty, point it to itself. Otherwise, traverse to the last node, point last->next to newNode, and newNode->next to head.
  3. Display: Start from head and traverse until temp->next == head.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

class CircularLinkedList {
    Node* head;
public:
    CircularLinkedList() { head = NULL; }
    
    void insert(int val) {
        Node* newNode = new Node();
        newNode->data = val;
        if (head == NULL) {
            head = newNode;
            newNode->next = head;
        } else {
            Node* temp = head;
            while (temp->next != head) temp = temp->next;
            temp->next = newNode;
            newNode->next = head;
        }
    }
    
    void display() {
        if (head == NULL) return;
        Node* temp = head;
        do {
            cout << temp->data << " -> ";
            temp = temp->next;
        } while (temp != head);
        cout << "(HEAD)\n";
    }
};

int main() {
    CircularLinkedList cll;
    cll.insert(100);
    cll.insert(200);
    cll.insert(300);
    cout << "Circular List: ";
    cll.display();
    return 0;
}

EX NO: 4 - Polynomial Subtraction using Singly Linked List

Aim

To write a C++ program to perform Polynomial Subtraction using a singly linked list.

Algorithm

  1. Define a Node with coefficient, power, and next pointer.
  2. Traverse both polynomials simultaneously.
  3. If power of poly1 > poly2, add poly1's term to result and advance poly1.
  4. If power of poly1 < poly2, subtract poly2's term, add to result, and advance poly2.
  5. If powers are equal, subtract coefficients, add to result, and advance both.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int coeff;
    int power;
    Node* next;
    Node(int c, int p) : coeff(c), power(p), next(NULL) {}
};

void insertNode(Node*& head, int c, int p) {
    Node* newNode = new Node(c, p);
    if (!head) head = newNode;
    else {
        Node* temp = head;
        while (temp->next) temp = temp->next;
        temp->next = newNode;
    }
}

void display(Node* head) {
    while (head) {
        cout << head->coeff << "x^" << head->power << (head->next ? " + " : "");
        head = head->next;
    }
    cout << endl;
}

Node* subtract(Node* p1, Node* p2) {
    Node* result = NULL;
    while (p1 && p2) {
        if (p1->power > p2->power) {
            insertNode(result, p1->coeff, p1->power); p1 = p1->next;
        } else if (p1->power < p2->power) {
            insertNode(result, -(p2->coeff), p2->power); p2 = p2->next;
        } else {
            insertNode(result, p1->coeff - p2->coeff, p1->power);
            p1 = p1->next; p2 = p2->next;
        }
    }
    while (p1) { insertNode(result, p1->coeff, p1->power); p1 = p1->next; }
    while (p2) { insertNode(result, -(p2->coeff), p2->power); p2 = p2->next; }
    return result;
}

int main() {
    Node *poly1 = NULL, *poly2 = NULL;
    insertNode(poly1, 5, 2); insertNode(poly1, 4, 1); insertNode(poly1, 2, 0); 
    insertNode(poly2, 3, 2); insertNode(poly2, 1, 1); 
    cout << "Poly 1: "; display(poly1);
    cout << "Poly 2: "; display(poly2);
    cout << "Result: "; display(subtract(poly1, poly2));
    return 0;
}

Output

Poly 1: 5x^2 + 4x^1 + 2x^0
Poly 2: 3x^2 + 1x^1
Result: 2x^2 + 3x^1 + 2x^0

Result

Thus, the program for polynomial subtraction was executed successfully and output verified.

EX NO: 5 - Queue Operations using a Linked List

Aim

To write a C++ program to implement Queue operations (Enqueue, Dequeue) using a linked list.

Algorithm

  1. Maintain two pointers: front and rear.
  2. Enqueue: Add a node at the rear end. If empty, both front and rear point to the new node.
  3. Dequeue: Remove the node from the front. If front becomes NULL, set rear to NULL.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
    Node(int val) : data(val), next(NULL) {}
};

class Queue {
    Node *front, *rear;
public:
    Queue() { front = rear = NULL; }
    
    void enqueue(int val) {
        Node* temp = new Node(val);
        if (rear == NULL) {
            front = rear = temp;
            return;
        }
        rear->next = temp;
        rear = temp;
    }
    
    void dequeue() {
        if (front == NULL) return;
        Node* temp = front;
        front = front->next;
        if (front == NULL) rear = NULL;
        cout << "Dequeued: " << temp->data << endl;
        delete temp;
    }
    
    void display() {
        Node* temp = front;
        while (temp != NULL) {
            cout << temp->data << " <- ";
            temp = temp->next;
        }
        cout << "REAR\n";
    }
};

int main() {
    Queue q;
    q.enqueue(10);
    q.enqueue(20);
    q.enqueue(30);
    q.display();
    q.dequeue();
    q.display();
    return 0;
}

Output

10 <- 20 <- 30 <- REAR
Dequeued: 10
20 <- 30 <- REAR

Result

Thus, the program to implement queue operations using a linked list was executed successfully.

EX NO: 6 - Circular Queue Operations using a Linked List

Aim

To write a C++ program to implement Circular Queue operations using a linked list.

Algorithm

  1. Maintain a rear pointer. The front is always rear->next.
  2. Enqueue: Insert after rear. If empty, newNode->next points to itself. Update rear.
  3. Dequeue: Remove node at rear->next. If it's the only node, set rear to NULL.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

class CircularQueue {
    Node* rear;
public:
    CircularQueue() { rear = NULL; }
    
    void enqueue(int val) {
        Node* temp = new Node();
        temp->data = val;
        if (rear == NULL) {
            rear = temp;
            rear->next = rear;
        } else {
            temp->next = rear->next;
            rear->next = temp;
            rear = temp;
        }
    }
    
    void dequeue() {
        if (rear == NULL) return;
        Node* temp = rear->next;
        if (rear == rear->next) {
            rear = NULL;
        } else {
            rear->next = temp->next;
        }
        cout << "Dequeued: " << temp->data << endl;
        delete temp;
    }
    
    void display() {
        if (rear == NULL) return;
        Node* temp = rear->next;
        do {
            cout << temp->data << " ";
            temp = temp->next;
        } while (temp != rear->next);
        cout << "(Circular)\n";
    }
};

int main() {
    CircularQueue cq;
    cq.enqueue(11);
    cq.enqueue(22);
    cq.display();
    cq.dequeue();
    cq.display();
    return 0;
}

Output

11 22 (Circular)
Dequeued: 11
22 (Circular)

Result

Thus, the program to implement a circular queue using a linked list was executed successfully.

EX NO: 7 - Deque Operations using a Doubly Linked List

Aim

To implement Deque (Double Ended Queue) operations using a doubly linked list.

Algorithm

  1. Use a Doubly Linked List with front and rear pointers.
  2. insertFront: Add node before the current front.
  3. insertRear: Add node after the current rear.
  4. deleteFront / deleteRear: Unlink from front/rear respectively.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int data;
    Node *prev, *next;
    Node(int val) : data(val), prev(NULL), next(NULL) {}
};

class Deque {
    Node *front, *rear;
public:
    Deque() { front = rear = NULL; }
    
    void insertFront(int val) {
        Node* temp = new Node(val);
        if (front == NULL) { front = rear = temp; }
        else { temp->next = front; front->prev = temp; front = temp; }
    }
    
    void insertRear(int val) {
        Node* temp = new Node(val);
        if (rear == NULL) { front = rear = temp; }
        else { temp->prev = rear; rear->next = temp; rear = temp; }
    }
    
    void display() {
        Node* temp = front;
        while (temp != NULL) {
            cout << temp->data << " ";
            temp = temp->next;
        }
        cout << endl;
    }
};

int main() {
    Deque dq;
    dq.insertRear(10);
    dq.insertRear(20);
    dq.insertFront(5);
    cout << "Deque: ";
    dq.display();
    return 0;
}

EX NO: 8 - Check Balanced Parentheses using Stack

Aim

To check if a given expression with parentheses is balanced using a stack.

Algorithm

  1. Traverse the expression string character by character.
  2. If an opening bracket is found, push to stack.
  3. If a closing bracket is found, pop from stack and check if it matches the bracket type.
  4. If the stack is empty at the end, the expression is balanced.

Program (C++)

#include <iostream>
#include <stack>
#include <string>
using namespace std;

bool isBalanced(string expr) {
    stack<char> s;
    for (int i = 0; i < expr.length(); i++) {
        if (expr[i] == '(' || expr[i] == '{' || expr[i] == '[') {
            s.push(expr[i]);
        } else if (expr[i] == ')' || expr[i] == '}' || expr[i] == ']') {
            if (s.empty()) return false;
            char top = s.top();
            s.pop();
            if ((expr[i] == ')' && top != '(') ||
                (expr[i] == '}' && top != '{') ||
                (expr[i] == ']' && top != '[')) {
                return false;
            }
        }
    }
    return s.empty();
}

int main() {
    string expr = "{[()]}";
    if (isBalanced(expr))
        cout << expr << " is Balanced\n";
    else
        cout << expr << " is Not Balanced\n";
    return 0;
}

Output

{[()]} is Balanced

Result

Thus, the balanced parentheses check using stack was executed successfully.

EX NO: 10 - Inheritance & Polymorphism

Aim

To demonstrate single inheritance and runtime polymorphism.

Algorithm

  1. Inheritance: Base class Shape holds dimensions. Derived class Triangle uses them.
  2. Polymorphism: Base class has a virtual function. Derived classes override it.

Program (C++)

#include <iostream>
using namespace std;

// (i) Inheritance
class Shape {
protected:
    double base, height;
public:
    void setValues(double b, double h) { base = b; height = h; }
};

class Triangle : public Shape {
public:
    double getArea() { return (base * height) / 2; }
};

// (ii) Polymorphism
class ShapeBase {
public:
    virtual void calculateArea() { cout << "Base Area" << endl; }
};

class Circle : public ShapeBase {
    float r = 5;
public:
    void calculateArea() override { cout << "Circle Area: " << 3.14 * r * r << endl; }
};

int main() {
    cout << "--- Inheritance ---\n";
    Triangle t;
    t.setValues(10, 5);
    cout << "Triangle Area: " << t.getArea() << endl;
    
    cout << "\n--- Polymorphism ---\n";
    ShapeBase* ptr;
    Circle c;
    ptr = &c;
    ptr->calculateArea(); 
    return 0;
}

EX NO: 11 - Infix Expression to Postfix Expression

Aim

To write a C++ program to convert an Infix expression into Postfix using a stack.

Algorithm

  1. Scan the string from left to right.
  2. If operand, output it. If (, push to stack. If ), pop to output until (.
  3. If operator, pop from stack to output while precedence of top >= current operator, then push current operator.

Program (C++)

#include <iostream>
#include <stack>
using namespace std;

int prec(char c) {
    if (c == '^') return 3;
    else if (c == '/' || c == '*') return 2;
    else if (c == '+' || c == '-') return 1;
    return -1;
}

void infixToPostfix(string s) {
    stack<char> st;
    string result;
    for (int i = 0; i < s.length(); i++) {
        char c = s[i];
        if (isalnum(c)) result += c;
        else if (c == '(') st.push('(');
        else if (c == ')') {
            while (st.top() != '(') { result += st.top(); st.pop(); }
            st.pop();
        } else {
            while (!st.empty() && prec(s[i]) <= prec(st.top())) {
                result += st.top(); st.pop();
            }
            st.push(c);
        }
    }
    while (!st.empty()) { result += st.top(); st.pop(); }
    cout << "Postfix: " << result << endl;
}

int main() {
    string exp = "A+B*C";
    cout << "Infix: " << exp << endl;
    infixToPostfix(exp);
    return 0;
}

Output

Infix: A+B*C
Postfix: ABC*+

Result

Thus, the infix to postfix conversion was executed successfully.

EX NO: 12 - Binary Search Tree (BST) Operations

Aim

To implement operations (Insertion and Inorder traversal) on a Binary Search Tree (BST) of integers.

Algorithm

  1. Define Node structure with key, left, right.
  2. Insert: If val < root, recur left. If val > root, recur right.
  3. Inorder: Traverse left subtree, print root, traverse right subtree.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int key;
    Node *left, *right;
    Node(int item) : key(item), left(NULL), right(NULL) {}
};

Node* insert(Node* node, int key) {
    if (node == NULL) return new Node(key);
    if (key < node->key) node->left = insert(node->left, key);
    else if (key > node->key) node->right = insert(node->right, key);
    return node;
}

void inorder(Node* root) {
    if (root != NULL) {
        inorder(root->left);
        cout << root->key << " ";
        inorder(root->right);
    }
}

int main() {
    Node* root = NULL;
    root = insert(root, 50);
    insert(root, 30);
    insert(root, 20);
    insert(root, 40);
    insert(root, 70);

    inorder(root);
    return 0;
}

EX NO: 13 - Search function for a B-Tree

Aim

To implement a basic search function for a B-Tree structure.

Algorithm

  1. Start search at root. Iterate through keys in the current node.
  2. If key matches, return true. If key is smaller, recurse down the corresponding child pointer.
  3. If leaf node is reached and not found, return false.

Program (C++)

#include <iostream>
using namespace std;

class BTreeNode {
public:
    int *keys; 
    BTreeNode **C; 
    int n; 
    bool leaf; 
    
    BTreeNode(int t, bool leaf_val) {
        keys = new int[2*t-1];
        C = new BTreeNode*[2*t];
        n = 0;
        leaf = leaf_val;
    }
    
    BTreeNode* search(int k) {
        int i = 0;
        while (i < n && k > keys[i]) i++;
        if (keys[i] == k) return this;
        if (leaf == true) return NULL;
        return C[i]->search(k);
    }
};

int main() {
    BTreeNode root(3, true);
    root.keys[0] = 10; root.keys[1] = 20; root.n = 2;
    
    cout << "Searching for 10 in B-Tree: " << (root.search(10) ? "Found" : "Not Found") << endl;
    cout << "Searching for 15 in B-Tree: " << (root.search(15) ? "Found" : "Not Found") << endl;
    return 0;
}

Output

Searching for 10 in B-Tree: Found
Searching for 15 in B-Tree: Not Found

Result

EX NO: 14 - Max-Heap Data Structure Operations

Aim

To implement operations (Insertion and Heapify) for a Max-Heap data structure.

Algorithm

  1. Use an array to store heap elements.
  2. Insert: Add element at end. Compare with parent and swap if it's greater. Repeat until root.

Program (C++)

#include <iostream>
#include <vector>
using namespace std;

class MaxHeap {
    vector<int> heap;
    
    void heapifyUp(int index) {
        if (index == 0) return;
        int parent = (index - 1) / 2;
        if (heap[parent] < heap[index]) {
            swap(heap[parent], heap[index]);
            heapifyUp(parent);
        }
    }
public:
    void insert(int val) {
        heap.push_back(val);
        heapifyUp(heap.size() - 1);
    }
    
    void display() {
        for (int i = 0; i < heap.size(); i++) cout << heap[i] << " ";
        cout << endl;
    }
};

int main() {
    MaxHeap h;
    h.insert(10); h.insert(30); h.insert(20); h.insert(40);
    cout << "Max Heap Array: ";
    h.display();
    return 0;
}

EX NO: 15 - AVL Tree Insertion Operation

Aim

To implement an AVL Tree insertion operation, maintaining balance factors.

Algorithm

  1. Perform standard BST insertion.
  2. Update the height of the current node.
  3. Get balance factor. If unbalanced, perform Left/Right rotations.

Program (C++)

#include <iostream>
using namespace std;

struct Node {
    int key, height;
    Node *left, *right;
    Node(int d) : key(d), height(1), left(NULL), right(NULL) {}
};

int height(Node *N) { return N ? N->height : 0; }
int getBalance(Node *N) { return N ? height(N->left) - height(N->right) : 0; }

Node *rightRotate(Node *y) {
    Node *x = y->left;
    Node *T2 = x->right;
    x->right = y; y->left = T2;
    y->height = max(height(y->left), height(y->right)) + 1;
    x->height = max(height(x->left), height(x->right)) + 1;
    return x;
}

Node *leftRotate(Node *x) {
    Node *y = x->right;
    Node *T2 = y->left;
    y->left = x; x->right = T2;
    x->height = max(height(x->left), height(x->right)) + 1;
    y->height = max(height(y->left), height(y->right)) + 1;
    return y;
}

Node* insert(Node* node, int key) {
    if (node == NULL) return new Node(key);
    if (key < node->key) node->left = insert(node->left, key);
    else if (key > node->key) node->right = insert(node->right, key);
    else return node;

    node->height = 1 + max(height(node->left), height(node->right));
    int balance = getBalance(node);

    if (balance > 1 && key < node->left->key) return rightRotate(node);
    if (balance < -1 && key > node->right->key) return leftRotate(node);

    return node;
}

void preOrder(Node *root) {
    if (root != NULL) {
        cout << root->key << " ";
        preOrder(root->left); preOrder(root->right);
    }
}

int main() {
    Node *root = NULL;
    root = insert(root, 10); root = insert(root, 20); root = insert(root, 30);
    cout << "Preorder of constructed AVL tree: ";
    preOrder(root);
    return 0;
}

EX NO: 16 - Hierarchical Inheritance & Runtime Polymorphism

Aim

To demonstrate hierarchical inheritance and runtime polymorphism.

Algorithm

  1. Hierarchical: Create one base class and multiple derived classes inheriting from the base class.
  2. Runtime Polymorphism: Use a base pointer pointing to a derived object to call a virtual function.

Program (C++)

#include <iostream>
using namespace std;

// (i) Hierarchical Inheritance
class Animal {
public:
    void eat() { cout << "Animal eats." << endl; }
};
class Dog : public Animal {
public:
    void bark() { cout << "Dog barks." << endl; }
};
class Cat : public Animal {
public:
    void meow() { cout << "Cat meows." << endl; }
};

// (ii) Runtime Polymorphism
class Base {
public:
    virtual void show() { cout << "Base Show" << endl; }
};
class Derived : public Base {
public:
    void show() override { cout << "Derived Show" << endl; }
};

int main() {
    cout << "--- Hierarchical Inheritance ---\n";
    Dog d; Cat c;
    d.eat(); d.bark();
    c.eat(); c.meow();

    cout << "\n--- Runtime Polymorphism ---\n";
    Base* bPtr;
    Derived dev;
    bPtr = &dev;
    bPtr->show(); 

    return 0;
}

Output

--- Hierarchical Inheritance ---
Animal eats.
Dog barks.
Animal eats.
Cat meows.

--- Runtime Polymorphism ---
Derived Show

Result

Thus, hierarchical inheritance and runtime polymorphism were executed successfully.
